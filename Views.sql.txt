CREATE VIEW CATEGORY_AUTHOR AS
SELECT C.category_name, A.FIRST_NAME || ' ' || A.LAST_NAME FULL_NAME
FROM CATEGORIES C
JOIN AUTHORS A ON C.CATEGORY_ID = A.CATEGORY_ID;

SELECT * FROM CATEGORY_AUTHOR;

CREATE VIEW DEFAULTER_MEMBER AS
SELECT M.FIRST_NAME , M.LAST_NAME, B.TITLE, B.PUBLISHER
FROM MEMBERS M
JOIN LOANS L ON L.MEMBER_ID = M.MEMBER_ID
JOIN BOOKS B ON B.BOOK_ID = L.BOOK_ID
WHERE RETURN_DATE IS NULL;


SELECT * FROM DEFAULTER_MEMBER;
